83 //
84 // Description for a circular buffer (also called "ring buffer")
85 // which is used as up-buffer (T->H)
86 //
87 typedef struct {
88  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
89  char* pBuffer; // Pointer to start of buffer
90  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
91  unsigned WrOff; // Position of next item to be written by either target.
92  volatile unsigned RdOff; // Position of next item to be read by host. Must be volatile since it may be modified by host.
93  unsigned Flags; // Contains configuration flags
94 } SEGGER_RTT_BUFFER_UP;
95 
96 //
97 // Description for a circular buffer (also called "ring buffer")
98 // which is used as down-buffer (H->T)
99 //
100 typedef struct {
101  const char* sName; // Optional name. Standard names so far are: "Terminal", "SysView", "J-Scope_t4i4"
102  char* pBuffer; // Pointer to start of buffer
103  unsigned SizeOfBuffer; // Buffer size in bytes. Note that one byte is lost, as this implementation does not fill up the buffer in order to avoid the problem of being unable to distinguish between full and empty.
104  volatile unsigned WrOff; // Position of next item to be written by host. Must be volatile since it may be modified by host.
105  unsigned RdOff; // Position of next item to be read by target (down-buffer).
106  unsigned Flags; // Contains configuration flags
107 } SEGGER_RTT_BUFFER_DOWN;
108 
109 //
110 // RTT control block which describes the number of buffers available
111 // as well as the configuration for each buffer
112 //
113 //
114 typedef struct {
115  char acID[16]; // Initialized to "SEGGER RTT"
116  int MaxNumUpBuffers; // Initialized to SEGGER_RTT_MAX_NUM_UP_BUFFERS (type. 2)
117  int MaxNumDownBuffers; // Initialized to SEGGER_RTT_MAX_NUM_DOWN_BUFFERS (type. 2)
118  SEGGER_RTT_BUFFER_UP aUp[SEGGER_RTT_MAX_NUM_UP_BUFFERS]; // Up buffers, transferring information up from target via debug probe to host
119  SEGGER_RTT_BUFFER_DOWN aDown[SEGGER_RTT_MAX_NUM_DOWN_BUFFERS]; // Down buffers, transferring information down from host via debug probe to target
120 } SEGGER_RTT_CB;
